CREATE PROCEDURE [dbo].[pFOBOfferList]
AS
BEGIN
	WITH CTE AS (
		SELECT
			pp.*,
			t.Name AS 'TermName',
			c.Name AS 'Name1',
			c.Sequance 'seqname',
			p.Name AS 'Name',
			s.Name as 'Name2',
			u.FullName AS 'UserName',
			cp.CropName AS 'CropName',
			RANK() OVER (PARTITION BY p.Name ORDER BY pp.PricingDate DESC) AS RowRank
		FROM [dbo].[tblProductPricing] pp
		INNER JOIN [dbo].[tblProduct] p ON p.ID = pp.ProductID
		INNER JOIN [dbo].[tblCategory] c ON c.ID = p.CategoryID
		INNER JOIN [dbo].[tblUser] u ON u.Id = pp.UserID
		LEFT JOIN [dbo].[tblShippingPeriod] s ON s.Id = pp.ShippingPeriodID
		INNER JOIN [dbo].[tblTerms] t ON pp.TermID = t.ID
		INNER JOIN [dbo].[tblCrop] cp ON pp.CropID = cp.ID
		where pp.IsActive = 1
		--WHERE p.Name LIKE '%Screen 15%' AND c.Name = 'Uganda- Robusta'
	)
	SELECT *, TermName AS 'TermName'
	FROM CTE
	WHERE RowRank = 1
	ORDER BY seqname, Name1;
END
